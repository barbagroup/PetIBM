120// Set initial conditions of the flow fields.
121PetscErrorCode SolutionSimple::setInitialConditions(const YAML::Node &node)
122{
123 PetscErrorCode ierr;
124 type::RealVec1D ICs;
125 std::vector<Vec> UGlobalUnpacked(dim);
126
127 PetscFunctionBeginUser;
128
129 // parse initial conditions for the velocity vector field
130 ierr = parser::parseICs(node, ICs); CHKERRQ(ierr);
131
132 // get individual velocity components from the packed Vec object
133 ierr = DMCompositeGetAccessArray(mesh->UPack, UGlobal, dim, nullptr,
134 UGlobalUnpacked.data()); CHKERRQ(ierr);
135
136 for (PetscInt i = 0; i < dim; ++i)
137 {
138 ierr = VecSet(UGlobalUnpacked[i], ICs[i]); CHKERRQ(ierr);
139 }
140
141 ierr = DMCompositeRestoreAccessArray(mesh->UPack, UGlobal, dim, nullptr,
142 UGlobalUnpacked.data());
143 CHKERRQ(ierr);
144
145 // pressure scalar field initialized with zeros
146 ierr = VecSet(pGlobal, 0.0); CHKERRQ(ierr);
147
148 PetscFunctionReturn(0);

151// Write flow field solutions to a file.
152PetscErrorCode SolutionSimple::write(const std::string &filePath) const
153{
154 PetscErrorCode ierr;
155
156 PetscFunctionBeginUser;
157
158 std::vector<Vec> vecs(dim + 1);
159 std::vector<std::string> names(dim + 1);
160
161 // set names for u, v, and/or w
162 for (int f = 0; f < dim; ++f) names[f] = type::fd2str[type::Field(f)];
163 // set name for pressure
164 names.back() = "p";
165
166 // get individual velocity components from the packed Vec object
167 ierr = DMCompositeGetAccessArray(mesh->UPack, UGlobal, dim, nullptr,
168 vecs.data()); CHKERRQ(ierr);
169 // get a reference to pressure Vec
170 vecs.back() = pGlobal;
171
172 // write to a HDF5 file
173 ierr = io::writeHDF5Vecs(comm, filePath, "/", names, vecs); CHKERRQ(ierr);
174
175 // nullify the reference to pressure Vec
176 vecs.back() = PETSC_NULL;
177 // return individual Vec objects to the packed Vec object
178 ierr = DMCompositeRestoreAccessArray(mesh->UPack, UGlobal, dim, nullptr,
179 vecs.data()); CHKERRQ(ierr);
180
181 PetscFunctionReturn(0);
